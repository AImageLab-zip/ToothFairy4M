  <name>Task 1: Implement user interactions in MaxilloNiiVueViewer</name>
  <files>static/js/modality_viewers/maxillo_niivue_viewer.js</files>
  <action>
Add user interaction handlers to MaxilloNiiVueViewer matching cbct.js behavior.

1. Add `_setupInteractions(orientation, canvas)` method:
   - Called after each NiiVue viewer is initialized
   - Attach event listeners to the canvas element

2. Implement scroll handling (wheel event):
   ```javascript
   canvas.addEventListener('wheel', (event) => {
       event.preventDefault();
       event.stopPropagation();

       const viewer = this.viewers[orientation];
       if (!viewer || !viewer.isReady()) return;

       if (event.ctrlKey) {
           // Ctrl+scroll = zoom
           const zoomDelta = event.deltaY > 0 ? -0.1 : 0.1;
           this._handleZoom(orientation, zoomDelta);
       } else {
           // Normal scroll = slice navigation
           const sliceDelta = event.shiftKey ? 5 : 1;  // Shift = fast scroll
           const direction = event.deltaY > 0 ? sliceDelta : -sliceDelta;
           this._handleSliceScroll(orientation, direction);
       }
   }, { passive: false });
   ```

3. Implement `_handleSliceScroll(orientation, direction)`:
   - Get current slice index from viewer
   - Calculate new index (clamped to valid range)
   - Set new slice index
   - Synchronization will auto-trigger via onSliceChange

4. Implement `_handleZoom(orientation, zoomDelta)`:
   - Use NiiVue's scene.pan2Dxyzmm for zoom control
   - Store zoom level per orientation in this.zoomLevels
   - Clamp between 1.0 and 5.0
   - Call viewer.nv.drawScene() after zoom change

5. Implement pan handling (left-click drag):
   ```javascript
   let isDragging = false;
   let lastX = 0, lastY = 0;

   canvas.addEventListener('mousedown', (event) => {
       if (event.button === 0) {
           isDragging = true;
           lastX = event.clientX;
           lastY = event.clientY;
           canvas.style.cursor = 'move';
       }
   });

   canvas.addEventListener('mousemove', (event) => {
       if (!isDragging) return;
       const deltaX = event.clientX - lastX;
       const deltaY = event.clientY - lastY;
       this._handlePan(orientation, deltaX, deltaY);
       lastX = event.clientX;
       lastY = event.clientY;
   });

   canvas.addEventListener('mouseup', () => {
       isDragging = false;
       canvas.style.cursor = 'crosshair';
   });
   ```

6. Implement `_handlePan(orientation, deltaX, deltaY)`:
   - Use NiiVue's pan2Dxyzmm property for pan offset
   - Only allow pan when zoomed in (zoom > 1.0)
   - Call viewer.nv.drawScene() after pan change

7. Add `resetAllViews()` method:
   - Reset zoom to 1.0 for all orientations
   - Reset pan to center
   - Reset slice positions to volume center
   - Update all views

1. Add `_updateSliceCounter(orientation)` method:

   ```javascript
   MaxilloNiiVueViewer.prototype._updateSliceCounter = function(orientation) {
       const viewer = this.viewers[orientation];
       if (!viewer || !viewer.isReady()) return;

       const containerId = this.containerPrefix + orientation + 'View';
       const container = document.getElementById(containerId);
       if (!container) return;

       let label = container.querySelector('.slice-counter');
       if (!label) {
           label = document.createElement('div');
           label.className = 'slice-counter';
           label.style.cssText = `
               position: absolute;
               bottom: 5px;
               right: 10px;
               color: white;
               font-size: 0.8rem;
               font-family: monospace;
               background: rgba(0, 0, 0, 0.7);
               padding: 2px 6px;
               border-radius: 3px;
               z-index: 100;
               pointer-events: none;
           `;
           container.appendChild(label);
       }

       const currentSlice = viewer.getSliceIndex() + 1;
       const totalSlices = viewer.getSliceCount();
       label.textContent = `${currentSlice} / ${totalSlices}`;
   };
   ```
