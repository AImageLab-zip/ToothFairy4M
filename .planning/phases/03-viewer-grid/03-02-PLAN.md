**Drag-drop initialization:**
```javascript
initDragDrop: function() {
  // Make modality chips draggable
  document.querySelectorAll('.modality-chip').forEach(chip => {
    chip.draggable = true;
    chip.addEventListener('dragstart', (e) => {
      const modality = chip.dataset.modality;
      const fileId = chip.dataset.fileId;

      // Don't allow dragging unavailable modalities
      if (!fileId || chip.classList.contains('unavailable')) {
        e.preventDefault();
        return;
      }

      e.dataTransfer.effectAllowed = 'copy';
      e.dataTransfer.setData('modality-slug', modality);
      e.dataTransfer.setData('file-id', fileId);

      chip.style.opacity = '0.5';
    });

    chip.addEventListener('dragend', (e) => {
      chip.style.opacity = '1';
    });
  });

  // Make windows drop zones
  document.querySelectorAll('.viewer-window').forEach((window, index) => {
    window.addEventListener('dragover', (e) => {
      e.preventDefault();
      e.dataTransfer.dropEffect = 'copy';
      window.classList.add('drag-over');
    });

    window.addEventListener('dragleave', (e) => {
      // Only remove if leaving window entirely (not entering child)
      if (!window.contains(e.relatedTarget)) {
        window.classList.remove('drag-over');
      }
    });

    window.addEventListener('drop', (e) => {
      e.preventDefault();
      const modality = e.dataTransfer.getData('modality-slug');
      const fileId = e.dataTransfer.getData('file-id');

      window.classList.remove('drag-over');

      if (modality && fileId) {
        this.loadModalityInWindow(index, modality, fileId);
      }
    });
  });
},
```

**Window state management:**
```javascript
loadModalityInWindow: async function(windowIndex, modality, fileId) {
  console.debug(`Loading ${modality} into window ${windowIndex}`);

  // Set loading state
  this.windowStates[windowIndex] = {
    modality: modality,
    loading: true,
    error: null,
    fileId: fileId
  };
  this.updateWindowUI(windowIndex);

  // Actual volume loading happens in Plan 03
  // For now, just update state to loaded after short delay (placeholder)
  setTimeout(() => {
    this.windowStates[windowIndex].loading = false;
    this.updateWindowUI(windowIndex);
  }, 500);
},

updateWindowUI: function(windowIndex) {
  const state = this.windowStates[windowIndex];
  const windowEl = document.querySelector(`[data-window-index="${windowIndex}"]`);

  if (!windowEl) return;

  if (state.loading) {
    // Show spinner
    windowEl.innerHTML = `
      <div class="spinner-border text-primary" role="status">
        <span class="visually-hidden">Loading...</span>
      </div>
    `;
    windowEl.classList.remove('loaded');
  } else if (state.error) {
    // Show error
    windowEl.innerHTML = `
      <div class="error-message text-danger">
        <i class="fas fa-exclamation-triangle"></i>
        <p>${state.error}</p>
        <button class="btn btn-sm btn-outline-primary" onclick="ViewerGrid.retryLoad(${windowIndex})">
          Retry
        </button>
      </div>
    `;
    windowEl.classList.remove('loaded');
  } else if (state.modality) {
    // Show loaded state with label
    // Viewer container will be added by Plan 03
    windowEl.innerHTML = `
      <div class="window-label">${state.modality.toUpperCase()}</div>
      <div class="viewer-container" id="viewer-window-${windowIndex}">
        <!-- Volume viewer will be initialized here in Plan 03 -->
        <div class="text-center text-white-50 p-4">
          <i class="fas fa-cube fa-3x mb-2"></i>
          <p>Viewer loading...</p>
        </div>
      </div>
    `;
    windowEl.classList.add('loaded');
  } else {
    // Empty state
    windowEl.innerHTML = `
      <div class="drop-hint">
        <i class="fas fa-arrow-down"></i>
        <p>Drop modality here</p>
      </div>
    `;
    windowEl.classList.remove('loaded');
  }
},

retryLoad: function(windowIndex) {
  const state = this.windowStates[windowIndex];
  if (state.fileId && state.modality) {
    this.loadModalityInWindow(windowIndex, state.modality, state.fileId);
  }
},
```

**Context menu for clearing windows:**
```javascript
initContextMenus: function() {
  document.querySelectorAll('.viewer-window').forEach((window, index) => {
    window.addEventListener('contextmenu', (e) => {
      e.preventDefault();

      // Only show menu if window has content
      if (!this.windowStates[index].modality) return;

      // Remove any existing context menus
      document.querySelectorAll('.viewer-context-menu').forEach(m => m.remove());

      // Create context menu
      const menu = document.createElement('div');
      menu.className = 'viewer-context-menu dropdown-menu show';
      menu.style.position = 'fixed';
      menu.style.left = e.clientX + 'px';
      menu.style.top = e.clientY + 'px';
      menu.style.zIndex = '1050';
      menu.innerHTML = `
        <button class="dropdown-item" onclick="ViewerGrid.clearWindow(${index})">
          <i class="fas fa-times me-2"></i>Clear
        </button>
      `;
      document.body.appendChild(menu);

      // Close menu on click outside
      const closeMenu = (event) => {
        if (!menu.contains(event.target)) {
          menu.remove();
          document.removeEventListener('click', closeMenu);
        }
      };
      setTimeout(() => {
        document.addEventListener('click', closeMenu);
      }, 0);
    });
  });
},
```
