```javascript
loadModalityInWindow: async function(windowIndex, modality, fileId) {
  console.debug(`Loading ${modality} (file ${fileId}) into window ${windowIndex}`);

  // Dispose existing viewer if present
  if (this.windowStates[windowIndex].viewerInstance) {
    try {
      this.windowStates[windowIndex].viewerInstance.dispose();
    } catch (e) {
      console.warn('Error disposing viewer:', e);
    }
  }

  // Set loading state
  this.windowStates[windowIndex] = {
    modality: modality,
    loading: true,
    error: null,
    fileId: fileId,
    viewerInstance: null
  };
  this.updateWindowUI(windowIndex);

  try {
    // FileRegistry API will handle authentication and path validation
    // The CBCTViewer expects the backend API endpoint, not direct file serving
    // Use the patient volume API which internally uses FileRegistry
    const scanId = window.djangoData?.scanId || window.scanId;
    const volumeUrl = `/${window.projectNamespace}/api/patient/${scanId}/volume/${modality}/`;

    // Create viewer container if not exists
    const windowEl = document.querySelector(`[data-window-index="${windowIndex}"]`);
    const viewerContainerId = `viewer-window-${windowIndex}`;
    let viewerContainer = document.getElementById(viewerContainerId);

    if (!viewerContainer) {
      windowEl.innerHTML = `
        <div class="window-label">${modality.toUpperCase()}</div>
        <div class="viewer-container" id="${viewerContainerId}">
          <div class="cbct-loading">
            <div class="text-center py-4">
              <div class="spinner-border text-primary" role="status">
                <span class="visually-hidden">Loading...</span>
              </div>
              <p class="mt-2 text-muted">Loading ${modality.toUpperCase()} data...</p>
            </div>
          </div>
          <div class="cbct-views" style="display: none;">
            <div class="row g-2">
              <div class="col-6">
                <div class="cbct-view-container">
                  <div class="cbct-view-label">Axial</div>
                  <div id="${viewerContainerId}_axialView" class="cbct-view"></div>
                </div>
              </div>
              <div class="col-6">
                <div class="cbct-view-container">
                  <div class="cbct-view-label">Sagittal</div>
                  <div id="${viewerContainerId}_sagittalView" class="cbct-view"></div>
                </div>
              </div>
              <div class="col-6">
                <div class="cbct-view-container">
                  <div class="cbct-view-label">Coronal</div>
                  <div id="${viewerContainerId}_coronalView" class="cbct-view"></div>
                </div>
              </div>
              <div class="col-6">
                <div class="cbct-view-container">
                  <div class="cbct-view-label">3D Volume</div>
                  <div id="${viewerContainerId}_volumeView" class="cbct-view"></div>
                </div>
              </div>
            </div>
          </div>
        </div>
      `;
      windowEl.classList.add('loaded');
    }

    // Initialize CBCTViewer for this window
    // CBCTViewer currently manages single global instance - we need to create window-specific instances
    // For now, use existing CBCTViewer but target specific container IDs
    if (typeof window.CBCTViewer !== 'undefined') {
      // Temporarily override targetModality and container prefix
      const originalTargetModality = window.CBCTViewer.targetModality;

      window.CBCTViewer.targetModality = modality;
      window.CBCTViewer.containerPrefix = viewerContainerId;

      await window.CBCTViewer.init(modality);

      // Store viewer reference
      this.windowStates[windowIndex].viewerInstance = window.CBCTViewer;
      this.windowStates[windowIndex].loading = false;
      this.updateWindowUI(windowIndex);

      // Restore original target (for other windows)
      window.CBCTViewer.targetModality = originalTargetModality;
    } else {
      throw new Error('CBCTViewer not available');
    }

  } catch (error) {
    console.error(`Error loading ${modality} in window ${windowIndex}:`, error);
    this.windowStates[windowIndex].loading = false;
    this.windowStates[windowIndex].error = error.message || 'Failed to load volume';
    this.updateWindowUI(windowIndex);
  }
},
```

**Update clearWindow to dispose viewers:**
```javascript
clearWindow: function(windowIndex) {
  console.debug(`Clearing window ${windowIndex}`);

  // Dispose viewer instance if exists
  const viewerInstance = this.windowStates[windowIndex].viewerInstance;
  if (viewerInstance && typeof viewerInstance.dispose === 'function') {
    try {
      viewerInstance.dispose();
    } catch (e) {
      console.warn('Error disposing viewer:', e);
    }
  }

  this.windowStates[windowIndex] = {
    modality: null,
    loading: false,
    error: null,
    fileId: null,
    viewerInstance: null
  };
  this.updateWindowUI(windowIndex);

  document.querySelectorAll('.viewer-context-menu').forEach(m => m.remove());
},
```

**Why this approach:**
- Reuses existing CBCTViewer infrastructure (already handles NIfTI parsing, slice rendering)
- Uses patient volume API which internally queries FileRegistry by modality
- Creates window-specific container IDs for viewer isolation
- Disposes viewers properly to prevent WebGL context leaks

**Limitation:**
Current CBCTViewer is singleton (window.CBCTViewer). For true multi-window support, we'd need to refactor CBCTViewer to support multiple instances. For Phase 3 MVP, we accept this limitation - only one window can actively load at a time, but once loaded, all windows display correctly.

**Alternative approach for true multi-window (if needed):**
Create separate viewer instances per window:
```javascript
this.windowStates[windowIndex].viewerInstance = new CBCTViewerClass(modality, containerPrefix);
```
This would require refactoring cbct.js to be a class instead of singleton object. Defer to Phase 4 if multi-window simultaneity is critical.
