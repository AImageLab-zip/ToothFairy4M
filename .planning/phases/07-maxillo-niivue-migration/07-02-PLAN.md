  <files>templates/maxillo/patient_detail_content.html</files>
  <action>
Update the windowing slider event handlers to work with the new NiiVue-based viewer.

1. Find the windowing slider section (lines ~65-75 in current template):
   ```html
   <div class="dropdown-menu p-3 windowing-menu" style="min-width: 260px;">
       <div class="mb-2">
           <label class="form-label-sm mb-1">Min (%) <span id="windowMinValue">0</span></label>
           <input type="range" class="form-range" id="windowMinRange" min="0" max="100" step="1" value="0">
       </div>
       ...
   </div>
   ```

2. Update or add JavaScript to handle windowing changes:
   ```javascript
   // Windowing controls handler
   function setupWindowingControls() {
       const minRange = document.getElementById('windowMinRange');
       const maxRange = document.getElementById('windowMaxRange');
       const minLabel = document.getElementById('windowMinValue');
       const maxLabel = document.getElementById('windowMaxValue');

       if (!minRange || !maxRange) return;

       const updateWindowing = () => {
           const minVal = parseInt(minRange.value, 10);
           const maxVal = parseInt(maxRange.value, 10);

           // Update labels
           if (minLabel) minLabel.textContent = minVal;
           if (maxLabel) maxLabel.textContent = maxVal;

           // Apply to viewer (works with both old and new viewer)
           if (window.maxilloViewer && window.maxilloViewer.setWindowingFromPercent) {
               window.maxilloViewer.setWindowingFromPercent(minVal, maxVal);
           } else if (window.CBCTViewer && window.CBCTViewer.initialized) {
               // Legacy Three.js path
               window.CBCTViewer.windowPercentMin = minVal;
               window.CBCTViewer.windowPercentMax = maxVal;
               window.CBCTViewer._cachedWindowParams = null;
               window.CBCTViewer.updateSlice('axial');
               window.CBCTViewer.updateSlice('sagittal');
               window.CBCTViewer.updateSlice('coronal');
           }
       };

       minRange.addEventListener('input', updateWindowing);
       maxRange.addEventListener('input', updateWindowing);
   }

   // Call on page load
   document.addEventListener('DOMContentLoaded', setupWindowingControls);
   ```

3. Ensure the Reset button also works with the new viewer:
   ```javascript
   document.getElementById('resetCBCTView')?.addEventListener('click', () => {
       if (window.maxilloViewer && window.maxilloViewer.resetAllViews) {
           window.maxilloViewer.resetAllViews();
       } else if (window.CBCTViewer) {
           window.CBCTViewer.resetAllViews();
       }
   });
   ```
