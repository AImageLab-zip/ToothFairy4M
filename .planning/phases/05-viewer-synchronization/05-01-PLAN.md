```javascript
/**
 * Update synchronization group membership when a window's orientation changes
 * @param {number} windowIndex - The window being updated
 * @param {string} newOrientation - The new orientation ('axial', 'sagittal', 'coronal')
 */
function updateOrientationGroup(windowIndex, newOrientation) {
    const oldOrientation = windowStates[windowIndex].currentOrientation;

    // Remove from old group
    synchronizationGroups[oldOrientation] =
        synchronizationGroups[oldOrientation].filter(i => i !== windowIndex);

    // Add to new group if not already present
    if (!synchronizationGroups[newOrientation].includes(windowIndex)) {
        synchronizationGroups[newOrientation].push(windowIndex);
    }

    console.log(`Window ${windowIndex}: ${oldOrientation} → ${newOrientation}`,
                'Groups:', synchronizationGroups);
}
```

3. **Create getGroupConsensusSlice function** (returns current slice for a group):
```javascript
/**
 * Get the current consensus slice index for an orientation group
 * @param {string} orientation - The orientation to check
 * @returns {number} The slice index, or 0 if no viewers in group
 */
function getGroupConsensusSlice(orientation) {
    const groupWindows = synchronizationGroups[orientation];

    // Find first ready viewer in the group
    for (const windowIndex of groupWindows) {
        const viewer = windowStates[windowIndex].niivueInstance;
        if (viewer && viewer.isReady()) {
            return viewer.getSliceIndex();
        }
    }

    return 0; // Default to first slice if no viewers ready
}
```

```javascript
/**
 * Initialize synchronization event listener for cross-window slice updates
 */
function initSynchronization() {
    window.addEventListener('sliceIndexChanged', (event) => {
        const { windowIndex, newSliceIndex } = event.detail;
        const sourceOrientation = windowStates[windowIndex].currentOrientation;

        // Don't propagate if source window is in free-scroll mode
        if (freeScrollWindows[windowIndex]) {
            return;
        }

        // Get all windows in the same orientation group
        const groupWindows = synchronizationGroups[sourceOrientation];

        // Propagate slice change to all other windows in group (excluding source)
        groupWindows.forEach(targetIndex => {
            if (targetIndex !== windowIndex && !freeScrollWindows[targetIndex]) {
                const targetViewer = windowStates[targetIndex].niivueInstance;
                if (targetViewer && targetViewer.isReady()) {
                    targetViewer.setSliceIndex(newSliceIndex);
                }
            }
        });
    });
}
```

```javascript
// Attach Free Scroll toggle handler
const freeScrollBtn = windowEl.querySelector('.free-scroll-btn');
if (freeScrollBtn) {
    freeScrollBtn.addEventListener('click', (e) => {
        e.stopPropagation();

        const isCurrentlyFreeScroll = freeScrollWindows[windowIndex];

        if (isCurrentlyFreeScroll) {
            // Re-sync: Join the group and snap to its current slice
            const orientation = windowStates[windowIndex].currentOrientation;
            const groupSlice = getGroupConsensusSlice(orientation);

            if (viewer && viewer.isReady()) {
                viewer.setSliceIndex(groupSlice);
            }

            freeScrollWindows[windowIndex] = false;
            freeScrollBtn.classList.remove('free-scroll-active');
            freeScrollBtn.innerHTML = '<i class="fas fa-link"></i>';
            freeScrollBtn.title = 'Toggle Free Scroll';

            console.log(`Window ${windowIndex}: Re-synced to group slice ${groupSlice}`);
        } else {
            // Enter free-scroll mode
            freeScrollWindows[windowIndex] = true;
            freeScrollBtn.classList.add('free-scroll-active');
            freeScrollBtn.innerHTML = '<i class="fas fa-link-slash"></i>';
            freeScrollBtn.title = 'Re-sync with Group';

            console.log(`Window ${windowIndex}: Entered free-scroll mode`);
        }
    });
}
```
